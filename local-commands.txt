# Send a test alert payload to a local monitoring webhook. simulates a Prometheus
# alert system sending a high-memory warning to a backend application 
# 1. Define the header using the default key found in your auth.py
# $headers = @{"X-API-KEY" = "dev-secret-key"}

# 1. Define the payload
$payload = @{
    incident_id = "INC-001"
    status = "firing"
    alertname = "HighMemoryUsage"
    description = "Memory usage is critical"
    service = "payment-gateway"
    severity = "critical"
}

# 2. Send the request
Invoke-RestMethod -Uri http://localhost:8000/webhook/prometheus `
    -Method Post `
    -Headers @{"X-API-KEY" = "dev-secret-key"} `
    -Body ($payload | ConvertTo-Json) `
    -ContentType "application/json"
